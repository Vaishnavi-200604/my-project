let users = [];

export default function handler(req, res) {
  if (req.method === "POST") {
    const { name, age, email, qualifications, skills, strengths } = req.body;
    const newUser = { id: users.length + 1, name, age, email, qualifications, skills, strengths };
    users.push(newUser);
    res.status(201).json({ message: "âœ… User added!", user: newUser });
  } else if (req.method === "GET") {
    res.status(200).json(users);
  }
}

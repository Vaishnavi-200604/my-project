  <script>
    function addQualification() {
      const container = document.getElementById("qualifications");
      const newInput = document.createElement("div");
      newInput.className = "qualification-group";
      newInput.innerHTML = '<input type="text" name="qualification[]" placeholder="Enter qualification">';
      container.appendChild(newInput);
    }

    document.getElementById("jobForm").addEventListener("submit", function(e) {
      e.preventDefault();
      const selectedStrengths = Array.from(document.getElementById("strengths").selectedOptions).map(opt => opt.value);
      if (selectedStrengths.length > 3) {
        alert("Please select no more than 3 strengths.");
        return;
      }
      alert("Form submitted successfully!");

    });
